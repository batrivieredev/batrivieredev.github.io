Concepts avancés MySQL

Découvrez les fonctionnalités avancées de MySQL pour optimiser vos bases de données.

Transactions

-- Début de transaction
START TRANSACTION;

-- Opérations
UPDATE comptes SET solde = solde - 100
WHERE id = 1;

UPDATE comptes SET solde = solde + 100
WHERE id = 2;

-- Si tout est OK
COMMIT;

-- En cas d'erreur
ROLLBACK;

-- Niveaux d'isolation
SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
SET TRANSACTION ISOLATION LEVEL REPEATABLE READ;
SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;

Procédures stockées

DELIMITER //

CREATE PROCEDURE transfert_argent(
    IN compte_source INT,
    IN compte_dest INT,
    IN montant DECIMAL(10,2)
)
BEGIN
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Erreur de transfert';
    END;

    START TRANSACTION;
        UPDATE comptes
        SET solde = solde - montant
        WHERE id = compte_source;

        UPDATE comptes
        SET solde = solde + montant
        WHERE id = compte_dest;
    COMMIT;
END //

DELIMITER ;

-- Appel de la procédure
CALL transfert_argent(1, 2, 100.00);

Optimisation des performances

Indexation

-- Index simple
CREATE INDEX idx_email
ON utilisateurs(email);

-- Index composé
CREATE INDEX idx_nom_prenom
ON utilisateurs(nom, prenom);

-- Index unique
CREATE UNIQUE INDEX idx_unique_email
ON utilisateurs(email);

-- Analyse des index
EXPLAIN SELECT * FROM utilisateurs
WHERE email = 'test@example.com';

Optimisation des requêtes

-- Éviter SELECT *
SELECT id, nom, email
FROM utilisateurs;

-- Utiliser LIMIT
SELECT * FROM logs
ORDER BY date DESC LIMIT 100;

-- Jointures optimisées
SELECT u.nom, c.date
FROM utilisateurs u
    FORCE INDEX (idx_id)
    JOIN commandes c
    ON u.id = c.user_id
WHERE u.statut = 'actif';

Réplication

# Configuration Master (my.cnf)
[mysqld]
server-id = 1
log_bin = mysql-bin
binlog_format = ROW
sync_binlog = 1

# Configuration Slave (my.cnf)
[mysqld]
server-id = 2
relay-log = slave-relay-bin

# Sur le Master
SHOW MASTER STATUS;

# Sur le Slave
CHANGE MASTER TO
    MASTER_HOST='master_ip',
    MASTER_USER='replication_user',
    MASTER_PASSWORD='password',
    MASTER_LOG_FILE='mysql-bin.000001',
    MASTER_LOG_POS=123;

START SLAVE;
SHOW SLAVE STATUS\G

Partitionnement

-- Partitionnement par plage
CREATE TABLE ventes (
    id INT,
    date DATE,
    montant DECIMAL(10,2)
)
PARTITION BY RANGE (YEAR(date)) (
    PARTITION p2022 VALUES LESS THAN (2023),
    PARTITION p2023 VALUES LESS THAN (2024),
    PARTITION p2024 VALUES LESS THAN (2025),
    PARTITION pmax VALUES LESS THAN MAXVALUE
);

-- Partitionnement par liste
CREATE TABLE produits (
    id INT,
    categorie VARCHAR(50)
)
PARTITION BY LIST COLUMNS(categorie) (
    PARTITION p_tech VALUES IN ('Ordinateurs', 'Smartphones'),
    PARTITION p_mode VALUES IN ('Vêtements', 'Accessoires'),
    PARTITION p_autres VALUES IN ('Divers')
);
