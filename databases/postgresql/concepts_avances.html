PostgreSQL - Concepts Avancés

Découvrez les fonctionnalités avancées de PostgreSQL pour créer des applications complexes et performantes.

Transactions et niveaux d'isolation

-- Transaction basique
BEGIN;
UPDATE accounts SET balance = balance - 100 WHERE id = 1;
UPDATE accounts SET balance = balance + 100 WHERE id = 2;
COMMIT;

-- Niveaux d'isolation
BEGIN TRANSACTION ISOLATION LEVEL READ COMMITTED;
-- ou
BEGIN TRANSACTION ISOLATION LEVEL REPEATABLE READ;
-- ou
BEGIN TRANSACTION ISOLATION LEVEL SERIALIZABLE;

-- Gestion des deadlocks
BEGIN;
SET LOCAL lock_timeout = '10s';
UPDATE accounts SET balance = balance - 100 WHERE id = 1;
-- Si deadlock, la transaction échouera après 10s
COMMIT;

-- Savepoints
BEGIN;
UPDATE accounts SET balance = balance - 100 WHERE id = 1;
SAVEPOINT my_savepoint;
UPDATE accounts SET balance = balance + 100 WHERE id = 2;
-- En cas d'erreur
ROLLBACK TO my_savepoint;
COMMIT;

Fonctions stockées

-- Fonction simple
CREATE OR REPLACE FUNCTION get_user_posts(user_id INTEGER)
RETURNS TABLE (
    id INTEGER,
    title VARCHAR,
    content TEXT
) AS $$
BEGIN
    RETURN QUERY
    SELECT p.id, p.title, p.content
    FROM posts p
    WHERE p.user_id = $1;
END;
$$ LANGUAGE plpgsql;

-- Fonction avec gestion d'erreurs
CREATE OR REPLACE FUNCTION transfer_money(
    from_account_id INTEGER,
    to_account_id INTEGER,
    amount DECIMAL
) RETURNS BOOLEAN AS $$
DECLARE
    from_balance DECIMAL;
BEGIN
    -- Vérifier le solde
    SELECT balance INTO from_balance
    FROM accounts
    WHERE id = from_account_id
    FOR UPDATE;

    IF from_balance < amount THEN
        RAISE EXCEPTION 'Solde insuffisant';
    END IF;

    -- Effectuer le transfert
    UPDATE accounts
    SET balance = balance - amount
    WHERE id = from_account_id;

    UPDATE accounts
    SET balance = balance + amount
    WHERE id = to_account_id;

    RETURN TRUE;
EXCEPTION
    WHEN OTHERS THEN
        RAISE NOTICE 'Une erreur est survenue: %', SQLERRM;
        RETURN FALSE;
END;
$$ LANGUAGE plpgsql;

Triggers

-- Fonction trigger pour mise à jour timestamp
CREATE OR REPLACE FUNCTION update_timestamp()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_timestamp_trigger
    BEFORE UPDATE ON users
    FOR EACH ROW
    EXECUTE FUNCTION update_timestamp();

-- Trigger pour audit
CREATE TABLE audit_logs (
    id SERIAL PRIMARY KEY,
    table_name VARCHAR(50),
    record_id INTEGER,
    action VARCHAR(20),
    old_data JSONB,
    new_data JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE OR REPLACE FUNCTION audit_changes()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO audit_logs (
            table_name,
            record_id,
            action,
            new_data
        ) VALUES (
            TG_TABLE_NAME,
            NEW.id,
            'INSERT',
            row_to_json(NEW)::jsonb
        );
    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO audit_logs (
            table_name,
            record_id,
            action,
            old_data,
            new_data
        ) VALUES (
            TG_TABLE_NAME,
            NEW.id,
            'UPDATE',
            row_to_json(OLD)::jsonb,
            row_to_json(NEW)::jsonb
        );
    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO audit_logs (
            table_name,
            record_id,
            action,
            old_data
        ) VALUES (
            TG_TABLE_NAME,
            OLD.id,
            'DELETE',
            row_to_json(OLD)::jsonb
        );
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

Vues et vues matérialisées

-- Vue simple
CREATE VIEW active_users AS
SELECT id, name, email
FROM users
WHERE is_active = true;

-- Vue avec agrégation
CREATE VIEW user_stats AS
SELECT
    u.id,
    u.name,
    COUNT(p.id) as post_count,
    COUNT(DISTINCT c.id) as comment_count,
    MAX(p.created_at) as last_post_date
FROM users u
LEFT JOIN posts p ON p.user_id = u.id
LEFT JOIN comments c ON c.user_id = u.id
GROUP BY u.id, u.name;

-- Vue matérialisée
CREATE MATERIALIZED VIEW monthly_stats AS
SELECT
    DATE_TRUNC('month', created_at) as month,
    COUNT(*) as total_posts,
    COUNT(DISTINCT user_id) as unique_authors,
    AVG(LENGTH(content)) as avg_content_length
FROM posts
GROUP BY DATE_TRUNC('month', created_at)
WITH DATA;

-- Rafraîchissement vue matérialisée
REFRESH MATERIALIZED VIEW monthly_stats;

-- Vue matérialisée avec index
CREATE MATERIALIZED VIEW post_search AS
SELECT
    id,
    title,
    content,
    to_tsvector('english', title || ' ' || content) as document
FROM posts
WITH DATA;

CREATE INDEX idx_post_search ON post_search USING gin(document);

Partitionnement

-- Table partitionnée par date
CREATE TABLE logs (
    id SERIAL,
    created_at TIMESTAMP,
    level VARCHAR(20),
    message TEXT
) PARTITION BY RANGE (created_at);

-- Création des partitions
CREATE TABLE logs_2024_q1 PARTITION OF logs
    FOR VALUES FROM ('2024-01-01') TO ('2024-04-01');

CREATE TABLE logs_2024_q2 PARTITION OF logs
    FOR VALUES FROM ('2024-04-01') TO ('2024-07-01');

-- Partitionnement par liste
CREATE TABLE sales (
    id SERIAL,
    amount DECIMAL,
    region VARCHAR(50)
) PARTITION BY LIST (region);

CREATE TABLE sales_europe PARTITION OF sales
    FOR VALUES IN ('FR', 'DE', 'IT', 'ES');

CREATE TABLE sales_asia PARTITION OF sales
    FOR VALUES IN ('CN', 'JP', 'KR', 'IN');

Extensions

-- PostGIS pour données géographiques
CREATE EXTENSION postgis;

SELECT ST_Distance(
    ST_GeomFromText('POINT(2.3522 48.8566)'),
    ST_GeomFromText('POINT(2.2945 48.8584)')
) as distance;

-- Full-text search avec pg_trgm
CREATE EXTENSION pg_trgm;

CREATE INDEX idx_users_name_trgm ON users
USING gin (name gin_trgm_ops);

SELECT name, similarity(name, 'John')
FROM users
WHERE name % 'John'
ORDER BY similarity DESC;

-- UUID pour identifiants uniques
CREATE EXTENSION "uuid-ossp";

CREATE TABLE documents (
    id UUID DEFAULT uuid_generate_v4(),
    title VARCHAR(255),
    content TEXT,
    PRIMARY KEY (id)
);
