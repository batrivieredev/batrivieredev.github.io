# library/my_module.py
from ansible.module_utils.basic import AnsibleModule

def main():
    module = AnsibleModule(
        argument_spec=dict(
            name=dict(required=True, type='str'),
            state=dict(default='present', choices=['present', 'absent']),
            value=dict(type='str')
        ),
        supports_check_mode=True
    )

    name = module.params['name']
    state = module.params['state']
    value = module.params['value']

    result = dict(
        changed=False,
        original_message='',
        message=''
    )

    if module.check_mode:
        module.exit_json(**result)

    # Logique du module
    try:
        if state == 'present':
            # Créer ou mettre à jour
            result['changed'] = True
            result['message'] = f'Created {name}'
        else:
            # Supprimer
            result['changed'] = True
            result['message'] = f'Removed {name}'
    except Exception as e:
        module.fail_json(msg=str(e))

    module.exit_json(**result)

if __name__ == '__main__':
    main()

# Callback plugin personnalisé
class CallbackModule(CallbackBase):
    CALLBACK_VERSION = 2.0
    CALLBACK_TYPE = 'notification'
    CALLBACK_NAME = 'custom'

    def v2_playbook_on_stats(self, stats):
        hosts = sorted(stats.processed.keys())
        for h in hosts:
            t = stats.summarize(h)
            if t['failures'] > 0 or t['unreachable'] > 0:
                self._notify_error(h, t)

    def _notify_error(self, host, stats):
        msg = f"Host {host} failed: {stats}"

# Filter plugin
def my_filter(value):
    return value.upper()

class FilterModule(object):
    def filters(self):
        return {
            'my_filter': my_filter
        }

# Lookup plugin
from ansible.plugins.lookup import LookupBase

class LookupModule(LookupBase):
    def run(self, terms, variables=None, **kwargs):
        return [self._lookup(term) for term in terms]

    def _lookup(self, term):
        # Logique de lookup
        return term.upper()

Dynamic inventory

#!/usr/bin/env python

import json
import subprocess

def get_ec2_instances():
    instances = {
        'webservers': {
            'hosts': [],
            'vars': {
                'http_port': 80
            }
        },
        'dbservers': {
            'hosts': [],
            'vars': {
                'db_port': 5432
            }
        },
        '_meta': {
            'hostvars': {}
        }
    }

    # AWS CLI pour lister les instances
    cmd = ['aws', 'ec2', 'describe-instances']
    result = subprocess.run(cmd, capture_output=True)
    ec2_data = json.loads(result.stdout)

    for reservation in ec2_data['Reservations']:
        for instance in reservation['Instances']:
            if instance['State']['Name'] != 'running':
                continue

            # Déterminer le groupe
            group = 'webservers'
            for tag in instance.get('Tags', []):
                if tag['Key'] == 'Role':
                    group = tag['Value'].lower()

            # Ajouter l'instance
            ip = instance.get('PublicIpAddress')
            if ip:
                instances[group]['hosts'].append(ip)
                instances['_meta']['hostvars'][ip] = {
                    'ansible_host': ip,
                    'instance_id': instance['InstanceId'],
                    'instance_type': instance['InstanceType']
                }

    return instances

def main():
    inventory = get_ec2_instances()
    print(json.dumps(inventory, indent=2))
