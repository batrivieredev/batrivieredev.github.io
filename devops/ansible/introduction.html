Introduction à Ansible

Ansible est un outil d'automatisation open-source qui permet de configurer des systèmes, déployer des logiciels et orchestrer des tâches plus complexes.

Structure de base

# Inventory (hosts)
[webservers]
web1.example.com
web2.example.com

[dbservers]
db1.example.com
db2.example.com

[production:children]
webservers
dbservers

# Variables d'inventaire
[webservers:vars]
http_port=80
proxy_timeout=5

# Playbook simple
- name: Configuration serveur web
  hosts: webservers
  become: true
  tasks:
    - name: Installation Nginx
      apt:
        name: nginx
        state: present

    - name: Démarrage service
      service:
        name: nginx
        state: started
        enabled: yes

    - name: Configuration
      template:
        src: nginx.conf.j2
        dest: /etc/nginx/nginx.conf
      notify: restart nginx

  handlers:
    - name: restart nginx
      service:
        name: nginx
        state: restarted

Commandes de base

# Exécution d'un playbook
ansible-playbook playbook.yml

# Avec inventaire spécifique
ansible-playbook -i inventory.ini playbook.yml

# Avec variables
ansible-playbook playbook.yml -e "version=1.0.0"

# Vérification (dry run)
ansible-playbook --check playbook.yml

# Exécution sur un hôte spécifique
ansible-playbook playbook.yml --limit web1.example.com

# Ad-hoc commands
ansible webservers -m ping
ansible all -m setup
ansible dbservers -m shell -a "free -m"

# Lister les hôtes
ansible all --list-hosts

# Vérifier la syntaxe
ansible-playbook --syntax-check playbook.yml

│   │   └── main.yml
│   ├── templates/
│   │   └── vhost.conf.j2
│   └── vars/
│       └── main.yml

# Utilisation dans un playbook
- name: Déploiement web
  hosts: webservers
  roles:
    - common
    - nginx
    - { role: app, tags: ['deploy'] }

# tasks/main.yml
- name: Installation des paquets
  apt:
    name: "{{ item }}"
    state: present
  loop:
    - nginx
    - certbot
    - python3-certbot-nginx

- name: Configuration HTTPS
  template:
    src: ssl.conf.j2
    dest: /etc/nginx/conf.d/ssl.conf
  notify: restart nginx

Variables et templates

  max_clients: 200

host_vars/
├── web1.yml

# Template Jinja2
server {
    listen {{ http_port }};
    server_name {{ domain_name }};

    location / {
        proxy_pass http://{{ app_host }}:{{ app_port }};
        proxy_set_header Host $host;
        proxy_cache_use_stale error timeout http_500;
    }

    {% if ssl_enabled %}
    listen 443 ssl;
    ssl_certificate {{ ssl_cert }};
    ssl_certificate_key {{ ssl_key }};
    {% endif %}
}

Configuration

# ansible.cfg
[defaults]
inventory = ./inventory
remote_user = ansible
private_key_file = ~/.ssh/ansible
roles_path = ./roles
host_key_checking = False
forks = 10

[privilege_escalation]
become = True
become_method = sudo
become_user = root
become_ask_pass = False

[ssh_connection]
pipelining = True
control_path = /tmp/ansible-ssh-%%h-%%p-%%r
ssh_args = -o ControlMaster=auto -o ControlPersist=60s

# Configuration des logs
[defaults]
log_path = /var/log/ansible.log
callback_whitelist = profile_tasks
bin_ansible_callbacks = True

# Cache de facts
fact_caching = jsonfile
fact_caching_connection = /tmp/ansible_facts
fact_caching_timeout = 7200
