Docker - Concepts Avancés

Découvrez les fonctionnalités avancées de Docker pour optimiser vos conteneurs et améliorer la sécurité.

Multi-stage Builds

# Build stage
FROM node:18 AS builder

WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
RUN npm run build

# Production stage
FROM nginx:alpine

# Copier uniquement les fichiers de build
COPY --from=builder /app/dist /usr/share/nginx/html

EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]

Avantages :

  • Images finales plus petites
  • Séparation des dépendances de build
  • Meilleure sécurité
  • Cache plus efficace

Sécurité des conteneurs

# Utiliser un utilisateur non-root
FROM node:18-alpine

# Créer un utilisateur
RUN addgroup -S appgroup && \
    adduser -S appuser -G appgroup

# Définir les permissions
WORKDIR /app
COPY --chown=appuser:appgroup . .

# Utiliser l'utilisateur non-root
USER appuser

# Scanner de vulnérabilités
# docker scan mon-image:latest

# Limiter les ressources
docker run -d \
  --memory="512m" \
  --cpus="1.0" \
  --pids-limit=100 \
  mon-image:latest

Réseaux avancés

# Créer un réseau overlay
docker network create \
  --driver overlay \
  --attachable \
  --subnet=10.0.0.0/24 \
  mon-reseau-overlay

# Configuration des DNS personnalisés
docker run -d \
  --name mon-app \
  --dns=8.8.8.8 \
  --dns-search=exemple.com \
  mon-image

# Macvlan pour accès direct au réseau
docker network create \
  -d macvlan \
  --subnet=192.168.1.0/24 \
  --gateway=192.168.1.1 \
  -o parent=eth0 \
  mon-reseau-macvlan

Docker Swarm

# Initialiser un swarm
docker swarm init --advertise-addr <IP>

# Déployer un service
docker service create \
  --name mon-service \
  --replicas 3 \
  --publish 80:80 \
  mon-image:latest

# Configuration du service
version: '3.8'
services:
  web:
    image: mon-image:latest
    deploy:
      replicas: 3
      update_config:
        parallelism: 1
        delay: 10s
      restart_policy:
        condition: on-failure
        max_attempts: 3
      resources:
        limits:
          cpus: '0.5'
          memory: 512M
    networks:
      - overlay-net

networks:
  overlay-net:
    driver: overlay

Secrets et Configs

# Créer un secret
echo "mon_mot_de_passe" | \
  docker secret create db_password -

# Utiliser un secret
docker service create \
  --name db \
  --secret db_password \
  postgres:14

# Configuration du service
services:
  db:
    image: postgres:14
    secrets:
      - source: db_password
        target: /run/secrets/db_password
    configs:
      - source: db_config
        target: /etc/postgres/conf.d/

secrets:
  db_password:
    external: true
configs:
  db_config:
    file: ./postgres.conf

Monitoring et Logging

# Collecter les métriques
docker stats

# Configuration du logging
docker run -d \
  --log-driver=json-file \
  --log-opts max-size=10m \
  --log-opts max-file=3 \
  mon-image

# Intégration avec ELK
version: '3.8'
services:
  app:
    image: mon-image
    logging:
      driver: "json-file"
      options:
        max-size: "10m"
        max-file: "3"
        labels: "production"
        env: "dev"

# Healthcheck
HEALTHCHECK --interval=30s \
            --timeout=3s \
            --start-period=5s \
            --retries=3 \
            CMD curl -f http://localhost/ || exit 1

Optimisation des images

# Utiliser .dockerignore
node_modules
npm-debug.log
Dockerfile
.git
.env

# Optimiser les layers
FROM node:18-alpine
WORKDIR /app

# Combiner les commandes RUN
RUN apk add --no-cache \
    python3 \
    make \
    g++ \
    && npm install -g npm@latest \
    && rm -rf /var/cache/apk/*

# Utiliser des images de base slim
FROM debian:slim-bullseye

# Cache des builds
docker build \
  --build-arg BUILDKIT_INLINE_CACHE=1 \
  --cache-from mon-image:latest \
  -t mon-image:nouvelle .
