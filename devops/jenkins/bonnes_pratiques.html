Jenkins - Bonnes Pratiques

Guide des meilleures pratiques pour créer et maintenir des pipelines Jenkins efficaces et sécurisés.

Structure du pipeline

✅ Bonnes pratiques

// Pipeline bien structuré
pipeline {
    agent any

    options {
        timeout(time: 1, unit: 'HOURS')
        buildDiscarder(logRotator(numToKeepStr: '10'))
        timestamps()
    }

    environment {
        APP_ENV = 'production'
    }

    stages {
        stage('Build') {
            steps {
                script {
                    // Une tâche par étape
                    buildApp()
                }
            }
        }
    }

    post {
        always {
            cleanWs()
        }
    }
}

❌ À éviter

// Mauvaises pratiques
node {
    // Scripts shell directement ❌
    sh """
        npm install
        npm test
        npm build
    """

    // Pas de gestion d'erreurs ❌
    if (env.BRANCH_NAME == 'main') {
        deploy()
    }
}

Sécurité

// Gestion des secrets
pipeline {
    environment {
        // ✅ Utiliser les credentials
        DOCKER_CREDS = credentials('docker-registry')
        AWS_CREDS = credentials('aws-keys')
    }

    stages {
        stage('Deploy') {
            steps {
                // ✅ withCredentials pour scope limité
                withCredentials([
                    string(
                        credentialsId: 'api-key',
                        variable: 'API_KEY'
                    )
                ]) {
                    sh './deploy.sh'
                }
            }
        }
    }
}

// ❌ Ne jamais exposer les secrets
environment {
    API_KEY = '1234567890' // ❌
}

// ❌ Éviter les commandes shell avec secrets
sh "docker login -u ${DOCKER_USER} -p ${DOCKER_PASS}" // ❌

Tests et qualité

pipeline {
    stages {
        stage('Quality') {
            parallel {
                // ✅ Tests en parallèle
                stage('Unit Tests') {
                    steps {
                        sh 'npm test'
                        junit '**/test-results.xml'
                    }
                }
                stage('Linting') {
                    steps {
                        sh 'npm run lint'
                        recordIssues(tools: [eslint()])
                    }
                }
                stage('Security Scan') {
                    steps {
                        sh 'npm audit'
                        dependencyCheck()
                    }
                }
            }
            // ✅ Toujours publier les résultats
            post {
                always {
                    publishCoverage()
                    recordIssues()
                }
            }
        }
    }

    // ✅ Archiver les artefacts importants
    post {
        success {
            archiveArtifacts artifacts: 'dist/**'
        }
    }
}

Organisation du code

// ✅ Structure recommandée
jenkins/
├── Jenkinsfile
├── vars/
│   ├── buildUtils.groovy
│   └── deployUtils.groovy
├── resources/
│   ├── docker/
│   │   └── Dockerfile
│   └── k8s/
│       └── deployment.yaml
└── scripts/
    ├── build.sh
    └── deploy.sh

// ✅ Fonctions réutilisables
// vars/buildUtils.groovy
def buildDocker(Map config) {
    docker.build(
        config.image,
        "--build-arg VERSION=${config.version} ."
    )
}

// Jenkinsfile
@Library('shared-lib') _

pipeline {
    stages {
        stage('Build') {
            steps {
                buildUtils.buildDocker(
                    image: 'myapp:latest',
                    version: '1.0.0'
                )
            }
        }
    }
}

Performance et maintenance

// ✅ Caching efficace
pipeline {
    options {
        // Réutiliser le workspace
        preserveStashes()
    }

    stages {
        stage('Build') {
            steps {
                // Cache des dépendances
                cache(path: '.npm') {
                    sh 'npm ci'
                }
            }
        }
    }
}

// ✅ Nettoyage régulier
properties([
    buildDiscarder(
        logRotator(
            numToKeepStr: '10',
            daysToKeepStr: '30',
            artifactNumToKeepStr: '5'
        )
    )
])

// ✅ Monitoring
pipeline {
    options {
        timeout(time: 1, unit: 'HOURS')
        timestamps()
        ansiColor('xterm')
    }

    post {
        always {
            // Nettoyer le workspace
            cleanWs()
            // Envoyer des métriques
            influxDbPublisher()
        }
    }
}

Conventions de nommage

  • Jobs : {projet}-{branche}-{action} (ex: myapp-main-build)
  • Stages : PascalCase (ex: BuildAndTest, DeployToStaging)
  • Variables : SCREAMING_SNAKE_CASE pour l'environnement
  • Fonctions : camelCase (ex: buildDocker, deployToKubernetes)
  • Credentials : {service}-{type}-{env} (ex: mysql-password-prod)
