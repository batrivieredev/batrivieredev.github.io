Introduction à Terraform

Terraform est un outil d'Infrastructure as Code (IaC) qui permet de créer, modifier et gérer l'infrastructure cloud de manière déclarative.

Configuration de base

# Provider configuration
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 4.0"
    }
  }
}

provider "aws" {
  region = "eu-west-3"
}

# Resources
resource "aws_instance" "web" {
  ami           = "ami-0c55b159cbfafe1f0"
  instance_type = "t2.micro"

  tags = {
    Name = "Web Server"
    Environment = "Production"
  }
}

# Variables
variable "instance_type" {
  type        = string
  default     = "t2.micro"
  description = "EC2 instance type"
}

# Outputs
output "instance_ip" {
  value = aws_instance.web.public_ip
}

Structure du projet

project/
├── main.tf         # Configuration principale
├── variables.tf    # Définition des variables
├── outputs.tf      # Valeurs de sortie
├── terraform.tfvars # Valeurs des variables
└── modules/
    ├── vpc/
    │   ├── main.tf
    │   ├── variables.tf
    │   └── outputs.tf
    └── ec2/
        ├── main.tf
        ├── variables.tf
        └── outputs.tf

# terraform.tfvars
region = "eu-west-3"
instance_type = "t2.micro"
environment = "production"

# Backend configuration
terraform {
  backend "s3" {
    bucket = "terraform-state"
    key    = "prod/terraform.tfstate"
    region = "eu-west-3"
  }
}

Commandes de base

# Initialisation
terraform init

# Vérification de la configuration
terraform plan

# Application des changements
terraform apply

# Destruction des ressources
terraform destroy

# Gestion de l'état
terraform show
terraform state list
terraform state show aws_instance.web

# Formatage et validation
terraform fmt
terraform validate

# Workspace
terraform workspace new dev
terraform workspace select prod
terraform workspace list

# Import de ressources existantes
terraform import aws_instance.web i-1234567890abcdef0

# Mise à jour des providers
terraform init -upgrade

Cycle de vie :

  1. Write (écriture de la configuration)
  2. Plan (vérification des changements)
  3. Apply (application des changements)
  4. Destroy (suppression si nécessaire)

Data Sources

# Récupération d'informations existantes
data "aws_vpc" "default" {
  default = true
}

data "aws_ami" "ubuntu" {
  most_recent = true

  filter {
    name   = "name"
    values = ["ubuntu/images/hvm-ssd/*"]
  }

  filter {
    name   = "virtualization-type"
    values = ["hvm"]
  }

  owners = ["099720109477"] # Canonical
}

# Utilisation
resource "aws_instance" "web" {
  ami           = data.aws_ami.ubuntu.id
  instance_type = var.instance_type
  subnet_id     = data.aws_vpc.default.id

  tags = {
    Name = "Web Server"
  }
}

Fonctions intégrées

# Manipulation de chaînes
locals {
  env_prefix = lower(var.environment)
  name       = format("%s-%s", var.project, local.env_prefix)
}

# Liste et maps
locals {
  instance_types = {
    dev  = "t2.micro"
    prod = "t2.small"
  }
  azs = slice(data.aws_availability_zones.available.names, 0, 3)
}

# Conditions
resource "aws_instance" "web" {
  instance_type = contains(["prod", "staging"], var.environment)
    ? "t2.small"
    : "t2.micro"

  tags = merge(
    var.common_tags,
    {
      Environment = var.environment
      Name        = local.name
    }
  )
}
