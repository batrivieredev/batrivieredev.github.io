Chart.js - Exemples Pratiques

Exemples pratiques d'utilisation de Chart.js pour différents scénarios de visualisation de données.

Dashboard de ventes

// Données de ventes
const salesData = {
  labels: [
    'Jan', 'Fév', 'Mar', 'Avr', 'Mai', 'Juin',
    'Juil', 'Août', 'Sep', 'Oct', 'Nov', 'Déc'
  ],
  datasets: [{
    label: 'Ventes 2024',
    data: [12000, 19000, 15000, 25000, 22000, 30000,
           28000, 27000, 32000, 35000, 38000, 42000],
    borderColor: 'rgb(75, 192, 192)',
    tension: 0.1,
    fill: true,
    backgroundColor: 'rgba(75, 192, 192, 0.2)'
  }, {
    label: 'Ventes 2023',
    data: [10000, 15000, 13000, 20000, 18000, 25000,
           24000, 23000, 28000, 30000, 32000, 35000],
    borderColor: 'rgb(201, 203, 207)',
    tension: 0.1,
    fill: true,
    backgroundColor: 'rgba(201, 203, 207, 0.2)'
  }]
};

// Configuration du graphique
const config = {
  type: 'line',
  data: salesData,
  options: {
    responsive: true,
    plugins: {
      title: {
        display: true,
        text: 'Évolution des ventes mensuelles'
      },
      tooltip: {
        mode: 'index',
        intersect: false,
        callbacks: {
          label: function(context) {
            let label = context.dataset.label || '';
            if (label) {
              label += ': ';
            }
            label += new Intl.NumberFormat('fr-FR', {
              style: 'currency',
              currency: 'EUR'
            }).format(context.parsed.y);
            return label;
          }
        }
      }
    },
    scales: {
      x: {
        display: true,
        title: {
          display: true,
          text: 'Mois'
        }
      },
      y: {
        display: true,
        title: {
          display: true,
          text: 'Ventes (€)'
        },
        ticks: {
          callback: function(value) {
            return new Intl.NumberFormat('fr-FR', {
              style: 'currency',
              currency: 'EUR'
            }).format(value);
          }
        }
      }
    }
  }
};

Répartition du budget

const budgetData = {
  labels: [
    'Marketing',
    'R&D',
    'RH',
    'Production',
    'Administration'
  ],
  datasets: [{
    data: [300000, 250000, 200000, 400000, 150000],
    backgroundColor: [
      'rgb(255, 99, 132)',
      'rgb(54, 162, 235)',
      'rgb(255, 206, 86)',
      'rgb(75, 192, 192)',
      'rgb(153, 102, 255)'
    ]
  }]
};

const budgetConfig = {
  type: 'doughnut',
  data: budgetData,
  options: {
    responsive: true,
    plugins: {
      legend: {
        position: 'right'
      },
      title: {
        display: true,
        text: 'Répartition du budget 2024'
      },
      tooltip: {
        callbacks: {
          label: function(context) {
            const value = context.parsed;
            const total = context.dataset.data.reduce((a, b) => a + b);
            const percentage = Math.round((value * 100) / total);
            return `${context.label}: ${percentage}% (${value}€)`;
          }
        }
      }
    }
  }
};
