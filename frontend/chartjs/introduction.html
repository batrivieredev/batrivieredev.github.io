Introduction à Chart.js

Chart.js est une bibliothèque JavaScript pour créer des graphiques interactifs et responsifs sur le web.

Installation et configuration

// Installation via npm
npm install chart.js

// Import
import Chart from 'chart.js/auto';

// Via CDN
<script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

// Configuration de base
const ctx = document.getElementById('myChart');
const chart = new Chart(ctx, {
  type: 'line',
  data: {
    labels: ['Jan', 'Fév', 'Mar', 'Avr', 'Mai'],
    datasets: [{
      label: 'Ventes 2024',
      data: [12, 19, 3, 5, 2],
      borderColor: 'rgb(75, 192, 192)',
      tension: 0.1
    }]
  },
  options: {
    responsive: true,
    plugins: {
      title: {
        display: true,
        text: 'Évolution des ventes'
      }
    }
  }
});

Mise à jour dynamique

// Ajout de données
function addData(chart, label, data) {
  chart.data.labels.push(label);
  chart.data.datasets.forEach((dataset) => {
    dataset.data.push(data);
  });
  chart.update();
}

// Suppression de données
function removeData(chart) {
  chart.data.labels.pop();
  chart.data.datasets.forEach((dataset) => {
    dataset.data.pop();
  });
  chart.update();
}

// Mise à jour en temps réel
setInterval(() => {
  const value = Math.random() * 100;
  const now = new Date();

  addData(
    chart,
    now.toLocaleTimeString(),
    value
  );

  // Limite à 10 points
  if (chart.data.labels.length > 10) {
    removeData(chart);
  }
}, 1000);

Événements et interactions

// Gestion des événements
chart.canvas.addEventListener('click', (evt) => {
  const points = chart.getElementsAtEventForMode(
    evt,
    'nearest',
    { intersect: true },
    false
  );

  if (points.length) {
    const point = points[0];
    const data = chart.data.datasets[point.datasetIndex]
      .data[point.index];
    console.log('Point cliqué:', data);
  }
});

// Configuration des tooltips personnalisés
options: {
  plugins: {
    tooltip: {
      callbacks: {
        label: function(context) {
          const label = context.dataset.label || '';
          const value = context.parsed.y;
          return `${label}: ${value}€`;
        },
        title: function(tooltipItems) {
          return 'Données pour ' +
            tooltipItems[0].label;
        }
      }
    }
  }
}

// Légendes interactives
options: {
  plugins: {
    legend: {
      onClick: function(e, legendItem, legend) {
        const index = legendItem.datasetIndex;
        const ci = legend.chart;

        ci.data.datasets[index].hidden =
          !ci.data.datasets[index].hidden;

        ci.update();
      }
    }
  }
}
