Blog avec commentaires

# Schéma
type Post {
  id: ID!
  title: String!
  content: String!
  author: User!
  comments: [Comment!]!
  tags: [String!]!
  createdAt: DateTime!
}

type User {
  id: ID!
  name: String!
  email: String!
  posts: [Post!]!
  comments: [Comment!]!
}

type Comment {
  id: ID!
  content: String!
  author: User!
  post: Post!
  createdAt: DateTime!
}

# Resolver
const resolvers = {
  Query: {
    posts: (parent, args, { db }) => db.posts.findAll(),
    post: (parent, { id }, { db }) => db.posts.findById(id),
    user: (parent, { id }, { db }) => db.users.findById(id)
  },

  Post: {
    author: (post, args, { db }) => db.users.findById(post.authorId),
    comments: (post, args, { db }) => db.comments.findByPostId(post.id)
  },

  Mutation: {
    createPost: async (parent, { input }, { db, currentUser }) => {
      if (!currentUser) throw new Error('Not authenticated');

      const post = await db.posts.create({
        ...input,
        authorId: currentUser.id
      });

      pubsub.publish('POST_CREATED', { postCreated: post });
      return post;
    },

    createComment: async (parent, { input }, { db, currentUser }) => {
      if (!currentUser) throw new Error('Not authenticated');

      const comment = await db.comments.create({
        ...input,
        authorId: currentUser.id
      });

      pubsub.publish('COMMENT_CREATED', { commentCreated: comment });
      return comment;
    }
  }
};

E-commerce avec panier

type Product {
  id: ID!
  name: String!
  description: String!
  price: Float!
  stock: Int!
  category: Category!
  images: [Image!]!
}

type CartItem {
  id: ID!
  product: Product!
  quantity: Int!
}

type Cart {
  id: ID!
  items: [CartItem!]!
  total: Float!
}

# Mutation
const resolvers = {
  Mutation: {
    addToCart: async (_, { productId, quantity }, { db, userId }) => {
      const product = await db.products.findById(productId);

      if (product.stock < quantity) {
        throw new Error('Not enough stock');
      }

      const cartItem = await db.cartItems.create({
        productId,
        quantity,
        userId
      });

      await db.products.update(productId, {
        stock: product.stock - quantity
      });

      return cartItem;
    },

    checkout: async (_, __, { db, userId }) => {
      const cart = await db.carts.findByUser(userId);
      const order = await db.orders.create({
        userId,
        items: cart.items,
        total: cart.total
      });

      await db.carts.clear(userId);
      return order;
    }
  }
};

Système d'authentification

# Types
type AuthPayload {
  token: String!
  user: User!
}

type Mutation {
  signup(input: SignupInput!): AuthPayload!
  login(input: LoginInput!): AuthPayload!
  resetPassword(input: ResetPasswordInput!): Boolean!
}

# Context
const context = async ({ req }) => {
  const token = req.headers.authorization;
  if (!token) return { };

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    const user = await db.users.findById(decoded.userId);
    return { currentUser: user };
  } catch (e) {
    throw new AuthenticationError('Invalid token');
  }
};

# Resolver
const resolvers = {
  Mutation: {
    signup: async (_, { input }, { db }) => {
      const existing = await db.users.findByEmail(input.email);
      if (existing) {
        throw new UserInputError('Email already exists');
      }

      const hashedPassword = await bcrypt.hash(
        input.password,
        10
      );

      const user = await db.users.create({
        ...input,
        password: hashedPassword
      });

      const token = jwt.sign(
        { userId: user.id },
        process.env.JWT_SECRET
      );

      return { token, user };
    },

    login: async (_, { input }, { db }) => {
      const user = await db.users.findByEmail(input.email);
      if (!user) {
        throw new AuthenticationError('Invalid credentials');
      }

      const valid = await bcrypt.compare(
        input.password,
        user.password
      );

      if (!valid) {
        throw new AuthenticationError('Invalid credentials');
      }

      const token = jwt.sign(
        { userId: user.id },
        process.env.JWT_SECRET
      );

      return { token, user };
    }
  }
};

Upload de fichiers

# Schéma
scalar Upload

type File {
  id: ID!
  filename: String!
  mimetype: String!
  encoding: String!
  url: String!
}

type Mutation {
  uploadFile(file: Upload!): File!
}

# Resolver
const resolvers = {
  Mutation: {
    uploadFile: async (_, { file }) => {
      const { createReadStream, filename, mimetype } =
        await file;

      const stream = createReadStream();
      const path = `uploads/${uuid()}-${filename}`;

      // Enregistrement local
      await new Promise((resolve, reject) => {
        stream
          .pipe(createWriteStream(path))
          .on('finish', resolve)
          .on('error', reject);
      });

      // Ou avec S3
      const upload = await s3.upload({
        Bucket: 'my-bucket',
        Key: path,
        Body: stream,
        ContentType: mimetype
      }).promise();

      return {
        filename,
        mimetype,
        url: upload.Location
      };
    }
  }
};

Gestion des erreurs

class ValidationError extends ApolloError {
  constructor(message) {
    super(message, 'VALIDATION_ERROR');

    Object.defineProperty(this, 'name', {
      value: 'ValidationError'
    });
  }
}

// Error formatting
const formatError = (error) => {
  if (error.originalError instanceof ValidationError) {
    return {
      message: error.message,
      code: error.extensions.code,
      field: error.extensions.field
    };
  }

  // Log les erreurs inattendues
  console.error(error);
  return {
    message: 'Internal server error',
    code: 'INTERNAL_ERROR'
  };
};

// Apollo Server config
const server = new ApolloServer({
  typeDefs,
  resolvers,
  formatError,
  context: ({ req }) => ({
    validate: (input, schema) => {
      const { error } = schema.validate(input);
      if (error) {
        throw new ValidationError(error.message);
      }
    }
  })
});
