Tailwind CSS - Concepts Avancés

Explorez les fonctionnalités avancées de Tailwind CSS pour créer des interfaces complexes et optimisées.

Plugins personnalisés

// plugins/aspect-ratio.js
const plugin = require('tailwindcss/plugin')

module.exports = plugin(function({ addUtilities, theme, variants }) {
  const ratios = {
    'aspect-square': {
      aspectRatio: '1 / 1'
    },
    'aspect-video': {
      aspectRatio: '16 / 9'
    },
    'aspect-portrait': {
      aspectRatio: '3 / 4'
    }
  }

  addUtilities(ratios, variants('aspectRatio'))
})

// tailwind.config.js
module.exports = {
  plugins: [
    require('./plugins/aspect-ratio'),
    require('@tailwindcss/forms'),
    require('@tailwindcss/typography')
  ]
}

Utilisation du plugin

<div class="aspect-video bg-gray-200">
  <img src="video-thumbnail.jpg" class="w-full h-full object-cover">
</div>

Configuration avancée des thèmes

// tailwind.config.js
const defaultTheme = require('tailwindcss/defaultTheme')

module.exports = {
  theme: {
    // Remplacement complet
    screens: {
      'tablet': '640px',
      'laptop': '1024px',
      'desktop': '1280px',
    },

    extend: {
      // Extension du thème par défaut
      colors: {
        primary: {
          50: '#f0f9ff',
          100: '#e0f2fe',
          500: '#0ea5e9',
          900: '#0c4a6e',
        }
      },

      fontSize: {
        'tiny': '.65rem',
        'mega': '4rem',
      },

      spacing: {
        '128': '32rem',
        'screen-10': '10vh',
      },

      animation: {
        'bounce-slow': 'bounce 3s linear infinite',
      },

      keyframes: {
        wiggle: {
          '0%, 100%': { transform: 'rotate(-3deg)' },
          '50%': { transform: 'rotate(3deg)' },
        }
      }
    }
  }
}

JIT (Just-In-Time) Mode

// tailwind.config.js
module.exports = {
  mode: 'jit',
  content: [
    './src/**/*.{html,js,vue,tsx}'
  ],

  theme: {
    extend: {
      // Valeurs arbitraires
      'width-[32.5%]': 'width: 32.5%',
      'rotate-[17deg]': 'transform: rotate(17deg)',
      'grid-cols-[1fr,500px,2fr]':
        'grid-template-columns: 1fr 500px 2fr'
    }
  }
}

// Utilisation
<div class="w-[32.5%] rotate-[17deg]">
  Contenu avec valeurs arbitraires
</div>

Directives avancées

/* styles.css */
@layer base {
  h1 {
    @apply text-2xl font-bold mb-4;
  }
  a {
    @apply text-blue-600 hover:text-blue-800;
  }
}

@layer components {
  .btn-primary {
    @apply py-2 px-4 bg-blue-500 text-white font-semibold rounded-lg shadow-md hover:bg-blue-700 focus:outline-none focus:ring-2 focus:ring-blue-400 focus:ring-opacity-75;
  }
}

@layer utilities {
  .scroll-snap-none {
    scroll-snap-type: none;
  }
  .scroll-snap-x {
    scroll-snap-type: x mandatory;
  }
}

/* Modification des variants */
@variants hover, focus, disabled {
  .custom-class {
    background-color: purple;
  }
}

Optimisation des performances

PurgeCSS Configuration

// tailwind.config.js
module.exports = {
  purge: {
    enabled: process.env.NODE_ENV === 'production',
    content: [
      './src/**/*.html',
      './src/**/*.vue',
      './src/**/*.jsx',
    ],
    options: {
      safelist: [
        /^bg-/,
        /^text-/,
        'rotate-90',
      ],
    },
  }
}
