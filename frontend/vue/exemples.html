Vue.js - Exemples Pratiques

Exemples de composants et fonctionnalités couramment utilisés dans les applications Vue.js.

Formulaire avec Validation

<script setup>
import { ref, reactive } from 'vue'
import { useField, useForm } from 'vee-validate'
import * as yup from 'yup'

const schema = yup.object({
  email: yup.string()
    .required('Email requis')
    .email('Email invalide'),
  password: yup.string()
    .required('Mot de passe requis')
    .min(8, 'Minimum 8 caractères'),
  confirmPassword: yup.string()
    .oneOf(
      [yup.ref('password')],
      'Les mots de passe doivent correspondre'
    )
})

const { handleSubmit, errors } = useForm({
  validationSchema: schema
})

const { value: email } = useField('email')
const { value: password } = useField('password')
const { value: confirmPassword } = useField('confirmPassword')

const onSubmit = handleSubmit((values) => {
  console.log('Form submitted:', values)
})
</script>

<template>
  <form @submit="onSubmit" class="form">
    <div class="form-group">
      <label>Email</label>
      <input
        v-model="email"
        type="email"
        class="form-control"
        :class="{ 'is-invalid': errors.email }"
      />
      <div class="invalid-feedback">
        {{ errors.email }}
      </div>
    </div>

    <div class="form-group">
      <label>Mot de passe</label>
      <input
        v-model="password"
        type="password"
        class="form-control"
        :class="{ 'is-invalid': errors.password }"
      />
      <div class="invalid-feedback">
        {{ errors.password }}
      </div>
    </div>

    <button type="submit" class="btn btn-primary">
      S'inscrire
    </button>
  </form>
</template>

Liste infinie avec Intersection Observer

<script setup>
import { ref, onMounted } from 'vue'

const items = ref([])
const loading = ref(false)
const page = ref(1)
const observer = ref(null)

const fetchItems = async () => {
  if (loading.value) return

  loading.value = true
  const response = await fetch(
    `/api/items?page=${page.value}`
  )
  const newItems = await response.json()

  items.value.push(...newItems)
  page.value++
  loading.value = false
}

const lastItemRef = (el) => {
  if (observer.value) {
    observer.value.disconnect()
  }

  observer.value = new IntersectionObserver(
    (entries) => {
      if (entries[0].isIntersecting) {
        fetchItems()
      }
    }
  )

  if (el) {
    observer.value.observe(el)
  }
}

onMounted(() => {
  fetchItems()
})
</script>

<template>
  <div class="items-list">
    <div
      v-for="(item, index) in items"
      :key="item.id"
      :ref="index === items.length - 1 ? lastItemRef : null"
      class="item"
    >
      {{ item.title }}
    </div>

    <div v-if="loading" class="loader">
      Chargement...
    </div>
  </div>
</template>

Tableau de données avec tri et filtres

<script setup>
const props = defineProps({
  data: Array,
  columns: Array
})

const sortBy = ref('')
const sortDir = ref('asc')
const searchQuery = ref('')

const sortedAndFilteredData = computed(() => {
  let result = [...props.data]

  // Filtrage
  if (searchQuery.value) {
    result = result.filter(item =>
      Object.values(item).some(val =>
        String(val)
          .toLowerCase()
          .includes(searchQuery.value.toLowerCase())
      )
    )
  }

  // Tri
  if (sortBy.value) {
    result.sort((a, b) => {
      const aVal = a[sortBy.value]
      const bVal = b[sortBy.value]

      if (sortDir.value === 'asc') {
        return aVal > bVal ? 1 : -1
      } else {
        return aVal < bVal ? 1 : -1
      }
    })
  }

  return result
})

const toggleSort = (column) => {
  if (sortBy.value === column) {
    sortDir.value = sortDir.value === 'asc'
      ? 'desc'
      : 'asc'
  } else {
    sortBy.value = column
    sortDir.value = 'asc'
  }
}
</script>

<template>
  <div>
    <input
      v-model="searchQuery"
      placeholder="Rechercher..."
      class="form-control mb-3"
    />

    <table class="table">
      <thead>
        <tr>
          <th
            v-for="col in columns"
            :key="col.key"
            @click="toggleSort(col.key)"
          >
            {{ col.label }}
            <span v-if="sortBy === col.key">
              {{ sortDir === 'asc' ? '↑' : '↓' }}
            </span>
          </th>
        </tr>
      </thead>
      <tbody>
        <tr v-for="item in sortedAndFilteredData">
          <td v-for="col in columns">
            {{ item[col.key] }}
          </td>
        </tr>
      </tbody>
    </table>
  </div>
</template>

Menu déroulant avec recherche

<script setup>
const props = defineProps({
  options: Array,
  modelValue: [String, Number],
  placeholder: String
})

const emit = defineEmits(['update:modelValue'])

const search = ref('')
const isOpen = ref(false)
const inputRef = ref(null)

const filteredOptions = computed(() => {
  if (!search.value) return props.options

  return props.options.filter(option =>
    option.label
      .toLowerCase()
      .includes(search.value.toLowerCase())
  )
})

const selectedOption = computed(() => {
  return props.options.find(
    option => option.value === props.modelValue
  )
})

const selectOption = (option) => {
  emit('update:modelValue', option.value)
  search.value = ''
  isOpen.value = false
}

onClickOutside(inputRef, () => {
  isOpen.value = false
})
</script>

<template>
  <div class="select" ref="inputRef">
    <div
      class="select-input"
      @click="isOpen = !isOpen"
    >
      <span v-if="selectedOption">
        {{ selectedOption.label }}
      </span>
      <span v-else class="placeholder">
        {{ placeholder }}
      </span>
    </div>

    <div v-show="isOpen" class="select-dropdown">
      <input
        v-model="search"
        placeholder="Rechercher..."
        @click.stop
      />

      <div class="options-list">
        <div
          v-for="option in filteredOptions"
          :key="option.value"
          class="option"
          :class="{
            'selected': option.value === modelValue
          }"
          @click="selectOption(option)"
        >
          {{ option.label }}
        </div>
      </div>
    </div>
  </div>
</template>
