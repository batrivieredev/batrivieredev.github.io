Introduction à Vue.js

Vue.js est un framework JavaScript progressif pour construire des interfaces utilisateur. Il est conçu pour être adopté de manière incrémentale.

Premier composant Vue

<template>
  <div class="counter">
    <h2>{{ title }}</h2>
    <p>Compteur: {{ count }}</p>
    <button @click="increment">
      Incrémenter
    </button>
  </div>
</template>

<script>
export default {
  name: 'Counter',
  data() {
    return {
      title: 'Mon Compteur',
      count: 0
    }
  },
  methods: {
    increment() {
      this.count++
    }
  }
}
</script>

<style scoped>
.counter {
  text-align: center;
  padding: 20px;
}
</style>

  • views/: Pages de l'application
  • router/: Configuration des routes
  • store/: Gestion de l'état (Vuex/Pinia)
  • assets/: Images, styles, etc.

Composition API

<script setup>
import { ref, computed, onMounted } from 'vue'

// État réactif
const count = ref(0)
const message = ref('Hello')

// Propriété calculée
const doubleCount = computed(() =>
  count.value * 2
)

// Méthodes
const increment = () => {
  count.value++
}

// Cycle de vie
onMounted(() => {
  console.log('Composant monté')
})
</script>

<template>
  <div>
    <p>{{ message }}</p>
    <p>Compteur: {{ count }}</p>
    <p>Double: {{ doubleCount }}</p>
    <button @click="increment">
      +1
    </button>
  </div>
</template>

Gestion d'État (Pinia)

// store/counter.js
import { defineStore } from 'pinia'

export const useCounterStore = defineStore('counter', {
  state: () => ({
    count: 0,
    history: []
  }),

  getters: {
    doubleCount: (state) => state.count * 2,
    lastUpdate: (state) => {
      return state.history[state.history.length - 1]
    }
  },

  actions: {
    increment() {
      this.count++
      this.history.push({
        type: 'increment',
        timestamp: new Date()
      })
    }
  }
})

// Utilisation dans un composant
import { useCounterStore } from '@/store/counter'

const counter = useCounterStore()
counter.increment()
console.log(counter.doubleCount)

Directives Vue

<template>
  <!-- Condition -->
  <div v-if="isVisible">Visible</div>
  <div v-else>Caché</div>

  <!-- Boucle -->
  <ul>
    <li v-for="item in items" :key="item.id">
      {{ item.name }}
    </li>
  </ul>

  <!-- Événements -->
  <button v-on:click="handleClick">
    Cliquer
  </button>

  <!-- Liaison de données -->
  <input v-model="message">
  <p>Message: {{ message }}</p>

  <!-- Classes dynamiques -->
  <div :class="{ active: isActive }">
    Élément
  </div>
</template>
