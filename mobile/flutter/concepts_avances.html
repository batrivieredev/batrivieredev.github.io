Gestion d'état avancée

// Riverpod
import 'package:flutter_riverpod/flutter_riverpod.dart';

final counterProvider = StateNotifierProvider((ref) {
  return Counter();
});

class Counter extends StateNotifier {
  Counter() : super(0);

  void increment() => state++;
  void decrement() => state--;
}

// BLoC Pattern
class CounterBloc extends Bloc {
  CounterBloc() : super(0) {
    on((event, emit) {
      emit(state + 1);
    });

    on((event, emit) {
      emit(state - 1);
    });
  }
}

// Utilisation
class CounterPage extends ConsumerWidget {
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final count = ref.watch(counterProvider);

    return Scaffold(
      body: Center(
        child: Text('$count'),
      ),
    );
  }
}

Animations complexes

// Animation contrôlée
class AnimatedLogo extends StatefulWidget {
  @override
  _AnimatedLogoState createState() => _AnimatedLogoState();
}

class _AnimatedLogoState extends State
    with SingleTickerProviderStateMixin {
  late AnimationController controller;
  late Animation animation;

  @override
  void initState() {
    super.initState();
    controller = AnimationController(
      duration: const Duration(seconds: 2),
      vsync: this,
    );

    animation = CurvedAnimation(
      parent: controller,
      curve: Curves.easeInOut,
    ).drive(Tween(begin: 0, end: 1));

    controller.forward();
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedBuilder(
      animation: animation,
      builder: (context, child) {
        return Transform.scale(
          scale: animation.value,
          child: child,
        );
      },
      child: FlutterLogo(size: 100),
    );
  }
}

Isolates et Performance

// Calcul intensif dans un Isolate
import 'dart:isolate';

Future> computeInBackground(List input) async {
  final receivePort = ReceivePort();

  await Isolate.spawn(heavyComputation, {
    'sendPort': receivePort.sendPort,
    'input': input,
  });

  return await receivePort.first;
}

void heavyComputation(Map params) {
  SendPort sendPort = params['sendPort'];
  List input = params['input'];

  // Calcul intensif
  final result = input.map((x) => x * x).toList();

  sendPort.send(result);
}

// Utilisation
void processData() async {
  final input = List.generate(10000, (i) => i);
  final result = await computeInBackground(input);
  print('Calcul terminé: $result');
}

class OptimizedImage extends StatelessWidget {

            color: Colors.white,
            width: 200,
            height: 200,

      cacheHeight: 200,
    );
  }

Plugins natifs

// Plugin Android (Kotlin)
class CustomPlugin : FlutterPlugin, MethodCallHandler {
  override fun onAttachedToEngine(binding: FlutterPlugin.FlutterPluginBinding) {
    val channel = MethodChannel(
      binding.binaryMessenger,
      "custom_plugin"
    )
    channel.setMethodCallHandler(this)
  }

  override fun onMethodCall(
    call: MethodCall,
    result: Result
  ) {
    when (call.method) {
      "getPlatformVersion" -> {
        result.success("Android ${android.os.Build.VERSION.RELEASE}")
      }
      else -> {
        result.notImplemented()
      }
    }
  }
}

// Plugin iOS (Swift)
class CustomPlugin: NSObject, FlutterPlugin {
  static func register(with registrar: FlutterPluginRegistrar) {
    let channel = FlutterMethodChannel(
      name: "custom_plugin",
      binaryMessenger: registrar.messenger()
    )
    let instance = CustomPlugin()
    registrar.addMethodCallDelegate(instance, channel: channel)
  }

  func handle(_ call: FlutterMethodCall,
             result: @escaping FlutterResult) {
    switch call.method {
    case "getPlatformVersion":
      result("iOS " + UIDevice.current.systemVersion)
    default:
      result(FlutterMethodNotImplemented)
    }
  }
}

Plateformes spécifiques

// Détection de plateforme
import 'dart:io' show Platform;

Widget getPlatformSpecificWidget() {
  if (Platform.isIOS) {
    return CupertinoApp(
      home: CupertinoPageScaffold(
        navigationBar: CupertinoNavigationBar(
          middle: Text('iOS App'),
        ),
        child: Center(
          child: CupertinoButton(
            child: Text('Bouton iOS'),
            onPressed: () {},
          ),
        ),
      ),
    );
  }

  return MaterialApp(
    home: Scaffold(
      appBar: AppBar(
        title: Text('Android App'),
      ),
      body: Center(
        child: ElevatedButton(
          child: Text('Bouton Android'),
          onPressed: () {},
        ),
      ),
    ),
  );
}

// Styles spécifiques
class PlatformStyle {
  static TextStyle get title {
    if (Platform.isIOS) {
      return CupertinoTheme.of(context)
        .textTheme
        .navLargeTitleTextStyle;
    }

    return Theme.of(context)
      .textTheme
      .headlineMedium!;
  }
}
