Bonnes pratiques de protection API

Guide des meilleures pratiques pour la sécurisation des API avec CORS et CSRF.

Configuration CORS sécurisée

// Mauvaise pratique ❌
app.use(cors());  // Autorise tout

// Bonne pratique ✅
const corsOptions = {
  origin: ['https://app.example.com'],
  methods: ['GET', 'POST', 'PUT', 'DELETE'],
  allowedHeaders: [
    'Content-Type',
    'Authorization'
  ],
  exposedHeaders: ['X-Total-Count'],
  credentials: true,
  maxAge: 3600,
  optionsSuccessStatus: 204
};

app.use(cors(corsOptions));

// Middleware personnalisé
const corsCheck = (req, res, next) => {
  const origin = req.headers.origin;
  const allowedOrigins = [
    'https://app.example.com',
    'https://admin.example.com'
  ];

  if (!origin || !allowedOrigins.includes(origin)) {
    return res.status(403).json({
      error: 'Origine non autorisée'
    });
  }

  // Vérification supplémentaire pour les admins
  if (origin === 'https://admin.example.com') {
    const token = req.headers.authorization;
    if (!isAdminToken(token)) {
      return res.status(403).json({
        error: 'Accès non autorisé'
      });
    }
  }

  next();
};

Protection CSRF robuste

// Mauvaise pratique ❌
app.disable('csrf');  // Désactive la protection

// Bonne pratique ✅
const csrf = require('csurf');
const cookieParser = require('cookie-parser');

app.use(cookieParser());
app.use(csrf({
  cookie: {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    sameSite: 'strict'
  }
}));

// Gestion personnalisée des erreurs CSRF
app.use((err, req, res, next) => {
  if (err.code === 'EBADCSRFTOKEN') {
    // Log de la tentative
    logger.warn('CSRF attempt', {
      ip: req.ip,
      path: req.path,
      headers: req.headers
    });

    return res.status(403).json({
      error: 'Session invalide'
    });
  }
  next(err);
});

Bonnes pratiques générales

// 1. Validation des entrées
const { body, validationResult } = require('express-validator');

const validateUser = [
  body('email')
    .trim()
    .isEmail()
    .normalizeEmail(),
  body('password')
    .trim()
    .isLength({ min: 8 }),
  (req, res, next) => {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      return res.status(400).json({
        errors: errors.array()
      });
    }
    next();
  }
];

// 2. Rate Limiting
const rateLimit = require('express-rate-limit');

const apiLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 100,
  message: 'Trop de requêtes'
});

// 3. Security Headers
const helmet = require('helmet');

app.use(helmet());
app.use(helmet.noSniff());
app.use(helmet.xssFilter());
app.use(helmet.hidePoweredBy());

// 4. Logging sécurisé
const winston = require('winston');

const logger = winston.createLogger({
  format: winston.format.json(),
  transports: [
    new winston.transports.File({
      filename: 'security.log',
      level: 'warn'
    })
  ]
});

À ne jamais faire :

  • ❌ Utiliser cors({ origin: '*' })
  • ❌ Désactiver la protection CSRF
  • ❌ Accepter tous les domaines
  • ❌ Négliger la validation des entrées
  • ❌ Exposer des informations sensibles

Sécurité des cookies

// Configuration des cookies
const session = require('express-session');

app.use(session({
  name: 'sessionId',
  secret: process.env.SESSION_SECRET,
  cookie: {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    sameSite: 'strict',
    domain: '.example.com',
    path: '/',
    maxAge: 3600000
  },
  resave: false,
  saveUninitialized: false
}));

// Rotation des secrets
const crypto = require('crypto');
const redis = require('redis');

class SessionManager {
  constructor() {
    this.redis = redis.createClient();
    this.secretRotationInterval = 24 * 60 * 60 * 1000;
  }

  async rotateSecret() {
    const newSecret = crypto
      .randomBytes(32)
      .toString('hex');

    await this.redis.set(
      'session_secret',
      newSecret,
      'EX',
      this.secretRotationInterval
    );

    return newSecret;
  }

  async getSecret() {
    let secret = await this.redis.get('session_secret');
    if (!secret) {
      secret = await this.rotateSecret();
    }
    return secret;
  }
}

Checklist de sécurité

1. Configuration CORS
  • ✅ Liste blanche d'origines
  • ✅ Méthodes HTTP limitées
  • ✅ Headers autorisés limités
  • ✅ Credentials sécurisés
2. Protection CSRF
  • ✅ Tokens par session
  • ✅ Validation à chaque requête
  • ✅ Gestion des erreurs
  • ✅ Logging des tentatives
3. Cookies
  • ✅ HttpOnly activé
  • ✅ Secure en production
  • ✅ SameSite strict
  • ✅ Domaine limité
4. Headers
  • ✅ CSP configuré
  • ✅ XSS Protection
  • ✅ No Sniff
  • ✅ HSTS activé
5. Monitoring
  • ✅ Rate limiting
  • ✅ Logging sécurisé
  • ✅ Alertes configurées
  • ✅ Audit régulier
